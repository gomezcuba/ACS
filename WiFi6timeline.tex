\frame{\frametitle{WiFi to WiFi6 in a Nutshell}

\begin{itemize}
 \item WiFi (.11g $\sim$ 56 Mbps) \\ \ \\

 \item WiFiN (.11n $\sim$ 600 Mbps)
 \begin{itemize}
    \item $40$ MHz bandwidth
    \item Single-user MIMO$\times 4$
    \item $5/6$ parity check\\ \ \\
\end{itemize}

 \item WiFi5 (.11.ac/ad $\sim$ 7 Gbps)
 \begin{itemize}
    \item $2\times 80$ MHz bandwidth
    \item MU-MIMO$\times 8$ BC only (synchronization)
    \item $256$-QAM\\ \ \\
\end{itemize}
 \item WiFi6 (.11.ax/ay $\sim$ 9.6 Gbps)
\begin{itemize}
 \item No more bandwidth or antennas (Spectral Efficiency)
 \item OFDMA and MAC MU-MIMO
 \item Assignment on top of contention DCF
 \item Emphasis on ``contention-free time reservations''\\ \ \\
\end{itemize}
\end{itemize}
